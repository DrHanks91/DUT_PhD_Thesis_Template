% 定义命令：将输入文本的每个字符（除空格外）替换为*
\newcommand{\starify}[1]{%
	\def\inputtext{#1}% 保存原始文本
	% 替换所有非空格字符为*（利用正则表达式匹配）
	\StrRegExpReplace{\inputtext}{[^\s]}{*}[\outputtext]%
	\outputtext% 输出结果
}
